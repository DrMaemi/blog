                  Node
                  template <class T>
class Node {
public:
    T val;
    Node* left = NULL;
    Node* right = NULL;
    
    bool operator<(const Node& rhs) const {
        return val < rhs.val;
    }

    Node(T val) {
        this->val = val;
    }
};
                  BinarySearchTree
                  template <class T>
class BinarySearchTree {
private:
    Node<T>* getPredecessor(Node<T>* cur) {
        Node<T>* node = cur->left;

        while (node->right != NULL) {
            node = node->right;
        }

        return node;
    }

    Node<T>* eraseRecursive(Node<T>* cur, const T& val) {
        if (cur == NULL)
            return NULL;
        
        if (val < cur->val) {
            cur->left = eraseRecursive(cur->left, val);

        }
        else if (cur->val < val)
            cur->right = eraseRecursive(cur->right, val);
        else {
            if (cur->left == NULL) {
                Node<T>* right = cur->right;
                free(cur);
                return right;
            }
            
            if (cur->right == NULL) {
                Node<T>* left = cur->left;
                free(cur);
                return left;
            }

            Node<T>* predecessor = getPredecessor(cur);
            cur->val = predecessor->val;
            cur->left = eraseRecursive(cur->left, predecessor->val);
        }

        return cur;
    }

    void inOrderTraversalRecursive(Node<T>* cur) {
        if (cur->left != NULL)
            inOrderTraversalRecursive(cur->left);

        cout << cur->val << " ";

        if (cur->right != NULL)
            inOrderTraversalRecursive(cur->right);
    }

public:
    Node<T>* root = NULL;

    BinarySearchTree() {}

    void insert(T val) {
        if (root == NULL) {
            root = new Node(val);
            cout << val << " inserted\n";
            return;
        }

        Node<T>* cur = root;
        bool isInserted = false;

        while (!isInserted) {
            if (val <= cur->val) {
                if (cur->left == NULL) {
                    cur->left = new Node(val);
                    isInserted = true;
                } else {
                    cur = cur->left;
                }
            } else {
                if (cur->right == NULL) {
                    cur->right = new Node(val);
                    isInserted = true;
                } else {
                    cur = cur->right;
                }
            }
        }

        cout << val << " inserted\n";
    }

    void erase(T val) {
        eraseRecursive(root, val);
        cout << val << " erased\n";
    }

    Node<T>* find(T val) {
        Node<T>* cur = root;
        
        while (cur != NULL) {
            if (val < cur->val) {
                cur = cur->left;
            } else if (cur->val < val) {
                cur = cur->right;
            } else {
                return cur;
            }
        }

        return NULL;
    }

    void inOrderTraversal() {
        if (root == NULL) {
            cout << "\n";
            return;
        }

        inOrderTraversalRecursive(root);
        cout << "\n";
    }
};
