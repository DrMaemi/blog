                                          # fish shell

                                          # 설치

                                          $ brew install fish

                                          $ which fish
/usr/local/bin/fish

                                          # 시스템이 사용 가능한 쉘 목록에 등록

                                          $ cat /etc/shells
/bin/bash
/bin/csh
/bin/dash
/bin/ksh
/bin/sh
/bin/tcsh
/bin/zsh

                                          # sudo -e /etc/shells or
$ sudo sh -c 'echo $(which fish) >> /etc/shells'
# List of acceptable shells for chpass(1).
# Ftpd will not allow users to connect who are not using
# one of these shells.
$ cat /etc/shells
/bin/bash
/bin/csh
/bin/dash
/bin/ksh
/bin/sh
/bin/tcsh
/bin/zsh
/usr/local/bin/fish

                                          # 시스템 기본 쉘로 변경

                                          $ chsh -s $(which fish)
