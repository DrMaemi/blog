                                                              struct Student {
    long long id;
    string name;

    bool operator>(const Student& rhs) const {
        if (id != rhs.id) return id > rhs.id;
        return name > rhs.name;
    }
};

struct StudentCompare {
    bool operator()(const Student& lhs, const Student& rhs) const {
        return lhs > rhs;
    }
};

set<Student, StudentCompare> s({
    {1, "1"},
    {2, "2"},
});
