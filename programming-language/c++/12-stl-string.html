                                                  vector<string> split(string& str) {
    vector<string> ret;
    string buffer = "";

    for (char x: str) {
        if (x == ' ') {
            if (!buffer.empty()) {
                ret.push_back(buffer);
            }
            buffer.clear();
        } else {
            buffer += x;
        }
    }

    if (!buffer.empty()) {
        ret.push_back(buffer);
    }
    return ret;
}
