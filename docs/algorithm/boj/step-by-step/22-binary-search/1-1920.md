## 코드
::: code-tabs
@tab C++
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int N;
    cin >> N;
    vector<int> nums(N);

    for (int i=0; i<N; i++) {
        cin >> nums[i];
    }

    sort(nums.begin(), nums.end());

    int M;
    cin >> M;
    for (int i=0; i<M; i++) {
        int n;
        cin >> n;
        cout << binary_search(nums.begin(), nums.end(), n) << "\n";
    }

    return 0;
}
```
